SELECT
UUT.UUT_SERIAL_NUMBER AS SERIAL_NUMBER,
convert(varchar,DATEADD(hour, 7, START_DATE_TIME),113) as Local_Date_Time,
UUT.TEST_SOCKET_INDEX,
SR.STEP_NAME,
DT.DATA

FROM [ATS_Results].[dbo].[PROP_RESULT] DT with (NOLOCK)
LEFT JOIN [ATS_Results].[dbo].[STEP_RESULT] SR with (NOLOCK)
ON SR.ID = DT.STEP_RESULT
LEFT JOIN [ATS_Results].[dbo].[UUT_RESULT] UUT with (NOLOCK)
ON UUT.ID = SR.UUT_RESULT
LEFT JOIN [ATS_Results].[dbo].[PROP_NUMERICLIMIT] LM with (NOLOCK)
ON LM.PROP_RESULT = DT.ID

Where SR.STEP_NAME not like ('%baseline%')
and START_DATE_TIME > GETDATE()-
