'
SELECT TOP 1 @start_time = START_DATE_TIME
FROM [ATSResults].[dbo].[PROP_RESULT] DT with (NOLOCK)
LEFT JOIN [ATSResults].[dbo].[STEP_RESULT] SR with (NOLOCK)
ON SR.ID = DT.STEP_RESULT
LEFT JOIN [ATSResults].[dbo].[UUT_RESULT] UUT with (NOLOCK)
ON UUT.ID = SR.UUT_RESULT
LEFT JOIN [ATSResults].[dbo].[PROP_NUMERICLIMIT] LM with (NOLOCK)
ON LM.PROP_RESULT = DT.ID 
WHERE START_DATE_TIME IS NOT NULL
and SR.STEP_NAME like ('%baseline%')
and STATION_ID = @STATIOIN
and TPS_NAME like ('%ATE CAL%')
and UUT.UUT_STATUS like ('%pass%')
and DT.PATH like N'%PARAMETER%' 
ORDER by UUT.START_DATE_TIME DESC ,SR.ORDER_NUMBER ASC

SELECT UUT.UUT_SERIAL_NUMBER AS SERIAL_NUMBER,
convert(varchar,DATEADD(hour, 7, START_DATE_TIME),113) as Local_Date_Time,
UUT.TEST_SOCKET_INDEX,
SR.STEP_NAME,
DT.DATA
FROM [ATSResults].[dbo].[PROP_RESULT] DT with (NOLOCK)
LEFT JOIN [ATSResults].[dbo].[STEP_RESULT] SR with (NOLOCK)
ON SR.ID = DT.STEP_RESULT
LEFT JOIN [ATSResults].[dbo].[UUT_RESULT] UUT with (NOLOCK)
ON UUT.ID = SR.UUT_RESULT
LEFT JOIN [ATSResults].[dbo].[PROP_NUMERICLIMIT] LM with (NOLOCK)
ON LM.PROP_RESULT = DT.ID 
WHERE START_DATE_TIME = @start_time
and SR.STEP_NAME like ('%baseline%')
and STATION_ID = @STATIOIN
and TPS_NAME like ('%ATE CAL%')
and UUT.UUT_STATUS like ('%pass%')
and DT.PATH like N'%PARAMETER%' 
ORDER by UUT.START_DATE_TIME DESC ,SR.ORDER_NUMBER ASC
